    const mensagens = [

        "Respire fundo, Maria Clara. O jornalismo precisa da sua sensibilidade. Você é capaz!",
        "Você é capaz de dominar qualquer pauta, não importa o quão difícil ela pareça agora.",

        "Você é capaz! A cada dia de estudo, você está mais perto de assinar sua primeira grande matéria."
    ];

    function gerarMensagem() {
        // Sorteia uma mensagem aleatória
        const randomIndex = Math.floor(Math.random() * mensagens.length);
        const display = document.getElementById('mensagem-display');
        
        // Efeito simples de fade in
        display.style.opacity = 0;
        setTimeout(() => {
            display.innerText = mensagens[randomIndex];
            display.style.transition = "opacity 0.5s";
            display.style.opacity = 1;
        }, 100);
    }
</script>
